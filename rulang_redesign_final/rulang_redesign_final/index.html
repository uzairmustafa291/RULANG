    🎯 RULang — Roman Urdu Programming Language
    
            
    
              RULang is a beginner-friendly programming language inspired by
              Roman Urdu keywords. It's designed as a compiler construction
              course project that demonstrates all 7 phases of compilation:
              Lexical Analysis, Syntax Analysis, Semantic Analysis, IR
              Generation, Optimization, Target Code Generation, and Code
              Emission.
            
    
          
    

          
    
            
    📝 COMPLETE LANGUAGE REFERENCE
    
            
    // ═══════════════════════════════════════════════════════════
// VARIABLES — Variables aur Constants
// ═══════════════════════════════════════════════════════════
rakho x = 10              // Number variable
rakho naam = "Ali"         // String variable
rakho flag = sach         // Boolean (sach=true, jhooth=false)
rakho n = 0

// ═══════════════════════════════════════════════════════════
// OUTPUT — Screen par dikhana
// ═══════════════════════════════════════════════════════════
likho "Salam Duniya!"      // String print karo
likho x                   // Variable ki value print karo
likho "Jama: " + x        // Concatenation

// ═══════════════════════════════════════════════════════════
// CONDITIONALS — Conditions aur Branching
// ═══════════════════════════════════════════════════════════
agar x > 5 to            // if condition
    likho "Bara hai"
khatam                   // End if block

agar x > 10 to           // if-else if-else
    likho "Bohat bada"
warna agar x == 5 to    // else if
    likho "Barabar"
warna                    // else
    likho "Chota"
khatam

// ═══════════════════════════════════════════════════════════
// SWITCH — Multiple Cases
// ═══════════════════════════════════════════════════════════
chunao din               // Switch on variable
    haalat 1
        likho "Somwar"
        todho            // Break ( IMPORTANT! )
    haalat 2
        likho "Mangal"
        todho
    warna                  // Default case
        likho "Aur din"
khatam

// ═══════════════════════════════════════════════════════════
// LOOPS — Repeat Karne ke liye
// ═══════════════════════════════════════════════════════════
// WHILE LOOP — Jabtak condition true hai
jabtak x < 10 karo       // while x < 10
    likho x
    rakho x = x + 1
khatam                   // End while

// DO-WHILE — Pehle ek baar, phir check
karo                     // do
    likho x
    rakho x = x + 1
jabtak x < 5             // while condition
khatam

// FOR LOOP — Count karne ke liye
gino i se 1 tak 10 karo  // for i from 1 to 10
    likho i
khatam

// FOR REVERSE — Ginti ulti tarike se
gino i se 10 tak 1 badhao -1 karo
    likho i
khatam

// FOR WITH STEP — Har dusra number
gino i se 0 tak 10 badhao 2 karo
    likho i
khatam

// ═══════════════════════════════════════════════════════════
// FUNCTIONS — Kaam jo dobara use hote hain
// ═══════════════════════════════════════════════════════════
kaam jama(a, b)         // Function declaration
    wapas a + b         // Return value
khatam

kaam factorial(n)        // Recursive function
    agar n <= 1 to
        wapas 1
    khatam
    wapas n * factorial(n - 1)
khatam

// Function call
likho jama(5, 3)
likho factorial(5)

// ═══════════════════════════════════════════════════════════
// BREAK & CONTINUE — Loop control
// ═══════════════════════════════════════════════════════════
jabtak x < 100 karo
    agar x == 50 to
        todho           // Loop se bahar nikal
    khatam
    likho x
khatam

jabtak i < 10 karo
    rakho i = i + 1
    agar i % 2 == 0 to
        agla            // Next iteration par jao
    khatam
    likho i            // Sirf odd numbers
khatam

// ═══════════════════════════════════════════════════════════
// LOGICAL OPERATORS — AND, OR, NOT
// ═══════════════════════════════════════════════════════════
agar x > 5 aur x < 10 to    // AND
    likho "Range mein"
khatam

agar x < 5 ya x > 10 to     // OR
    likho "Range ke bahar"
khatam

agar nahi (x == 0) to        // NOT
    likho "Zero nahi hai"
khatam

    🏗️ 7 LAYERS OF COMPILER
    
            
    Layer 1: LEXICAL ANALYSIS    → Scanner/Tokenizer
         Input: "rakho x = 10"
         Output: [RAKHO, IDENT, EQ, NUMBER]

Layer 2: SYNTAX ANALYSIS     → Parser
         Input: [RAKHO, IDENT, EQ, NUMBER]
         Output: AST (Abstract Syntax Tree)

Layer 3: SEMANTIC ANALYSIS  → Type & Scope Checking
         Builds Symbol Table, Type Inference

Layer 4: IR GENERATION      → Three-Address Code
         MOV t0, 10
         STORE x, t0

Layer 5: OPTIMIZATION      → Code Improvement
         Constant Folding, Dead Code Elimination

Layer 6: TARGET CODE        → Assembly
         MOV EAX, 10
         MOV [x], EAX

Layer 7: CODE EMISSION     → Machine Code
         B8 0A 00 00 00...
